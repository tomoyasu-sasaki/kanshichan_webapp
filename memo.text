
# 仮想環境を作成
python -m venv .kanshichan

# 仮想環境を有効化
source .kanshichan/bin/activate

# メインプログラムを実行
python -m src.kanshichan.main

# 仮想環境を無効化
deactivate

# 仮想環境を削除
rm -rf .kanshichan

# 依存関係をファイルに出力
pip freeze > requirements.txt

# 依存関係をインストール
pip install -r requirements.txt

# テストを実行
pytest

# テストカバレッジを出力
pytest --cov=monitor

# テストカバレッジをHTMLで表示
pytest --cov-report html

# ngrokを起動
ngrok http http://localhost:5001

# ポートを確認
lsof -i :5001

# ポートを解放
kill -9 <PID>
